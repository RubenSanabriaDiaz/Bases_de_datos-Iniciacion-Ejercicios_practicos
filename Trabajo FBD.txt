CREATE TABLE empresa(
              nif                  VARCHAR (9)           CONSTRAINT pk_empresa                      PRIMARY KEY,
              nombre               VARCHAR(30)           CONSTRAINT nn_nombre_empresa               NOT NULL);
              
CREATE TABLE emp_emp(
              nif_1       VARCHAR(9),
              nif_2       VARCHAR(9),
              CONSTRAINT pk_emp_emp PRIMARY KEY(nif_1, nif_2),
              CONSTRAINT fk_nif_1_emp_emp FOREIGN KEY(nif_1) REFERENCES empresa,
              CONSTRAINT fk_nif_2_emp_emp FOREIGN KEY(nif_2) REFERENCES empresa);
              
CREATE TABLE juego(
              id_juego             VARCHAR(7)            CONSTRAINT pk_juego                        PRIMARY KEY,
              nombre               VARCHAR(20)           CONSTRAINT nn_nombre_juego                 NOT NULL,
              f_publicacion        DATE,
              n_ventas             NUMBER                CONSTRAINT nn_n_ventas_juego               NOT NULL,
              precio               NUMBER(4,2)           CONSTRAINT ck_precio_jugo                  CHECK  (precio > 20 AND precio < 90),
              descripcion          VARCHAR(200)          CONSTRAINT nn_descripcion_juego            NOT NULL,
              tipo                 VARCHAR(15)           CONSTRAINT nn_tipo_juego                   NOT NULL,
              nif_1                VARCHAR(9)            CONSTRAINT fk_nif_1_juego                  REFERENCES empresa,
              nif_2                VARCHAR(9),
              nif_3                VARCHAR(9),
              CONSTRAINT fk_nif_2_nif_3_juego FOREIGN KEY(nif_2, nif_3) REFERENCES emp_emp);

CREATE TABLE actualizacion(
              version              VARCHAR(9),
              f_actualizacion      DATE,
              descripcion          VARCHAR(200)          CONSTRAINT nn_descripcion_actualizacion    NOT NULL,
              id_J                 VARCHAR(7)            CONSTRAINT fk_id_juego_actualizacion       REFERENCES juego,
              CONSTRAINT pk_version_id_J_actualizacion PRIMARY KEY(version, id_J));
            
CREATE TABLE cp(
              cp            NUMBER(5,0),
              localidad     VARCHAR(20)       CONSTRAINT      nn_localidad_cp     NOT NULL,
              CONSTRAINT pk_cp PRIMARY KEY(cp));            
              
CREATE TABLE empleado(
              dni                  VARCHAR (9)           CONSTRAINT pk_empleado                     PRIMARY KEY,
              nombre               VARCHAR (10)          CONSTRAINT nn_nombre_empleado              NOT NULL,
              apellidos            VARCHAR (20)          CONSTRAINT nn_apellidos_empleado           NOT NULL,
              f_nacimiento         DATE,
              domicilio            VARCHAR(30)           CONSTRAINT nn_domicilio_empleado           NOT NULL,
              cp                   NUMBER(5,0)           CONSTRAINT fk_cp_empleado                  REFERENCES cp,
              nif                  VARCHAR(9)            CONSTRAINT fk_nif_empleado                 REFERENCES empresa);

CREATE TABLE agencia(
              id_agencia           VARCHAR (9)           CONSTRAINT pk_agencia                      PRIMARY KEY,
              nombre               VARCHAR(30)           CONSTRAINT nn_nombre_agencia               NOT NULL);

CREATE TABLE producto(
             cod_pro               VARCHAR(7)            CONSTRAINT nn_cod_pro_producto             NOT NULL,
             nombre                VARCHAR(30)           CONSTRAINT nn_nombre_producto              NOT NULL,
             descripcion           VARCHAR(200)          CONSTRAINT nn_descripcion_producto         NOT NULL,
             precio                number(5,2)           CONSTRAINT ck_precio_producto              CHECK (precio > 10 AND precio < 400),
             nif                   VARCHAR(9)            CONSTRAINT fk_nif_producto                 REFERENCES empresa,
             CONSTRAINT pk_producto PRIMARY KEY(cod_pro, nif));
             
CREATE TABLE ag_em(
             id_agencia           VARCHAR(9)            CONSTRAINT fk_id_agencia_ag_em           REFERENCES agencia,
             nif                  VARCHAR(9)            CONSTRAINT fk_nif_ag_em                  REFERENCES empresa,
             CONSTRAINT pk_ag_em PRIMARY KEY(id_agencia, nif));

CREATE TABLE anuncio(
             id_anuncio           VARCHAR(7)            CONSTRAINT nn_version_anuncio               NOT NULL,
             duracion             NUMBER(3,0)           CONSTRAINT ck_duracion_anuncio              CHECK (duracion > 0 AND duracion < 180)
                                                        CONSTRAINT nn_duracion_anuncio              NOT NULL,      
             precio               NUMBER(10,2)          CONSTRAINT nn_precio_anuncio                NOT NULL,
             id_agencia           VARCHAR(9)            CONSTRAINT fk_id_agencia_anuncio            REFERENCES agencia,
             nif                  VARCHAR(9),         
             cod_pro              VARCHAR(7),
             id_juego             VARCHAR(7)            CONSTRAINT fk_id_juego_anuncio              REFERENCES juego,
             CONSTRAINT fK_id_agencia_nif_anuncio FOREIGN KEY(id_agencia, nif) REFERENCES ag_em,
             CONSTRAINT fk_nif_cod_pro_anuncio FOREIGN KEY(cod_pro, nif) REFERENCES producto,
             CONSTRAINT pk_anuncio PRIMARY KEY(id_anuncio, id_agencia, nif));
              
CREATE TABLE director(
              dni                  VARCHAR(9),
              CONSTRAINT pk_director PRIMARY KEY(dni),
              CONSTRAINT pk_dni_director FOREIGN KEY(dni) references empleado);

CREATE TABLE programador(
              dni_1         VARCHAR(9),
              dni_2         VARCHAR(9),
              CONSTRAINT pk_programador PRIMARY KEY(dni_1),
              CONSTRAINT fk_dni_1_programador FOREIGN KEY(dni_1) REFERENCES empleado,
              CONSTRAINT fk_dni_2_programador FOREIGN KEY(dni_2) REFERENCES director);
              
CREATE TABLE diseñador(
              dni_1         VARCHAR(9),
              dni_2         VARCHAR(9),
              CONSTRAINT pk_diseñador PRIMARY KEY(dni_1),
              CONSTRAINT fk_dni_1_diseñador FOREIGN KEY(dni_1) REFERENCES empleado,
              CONSTRAINT fk_dni_2_diseñador FOREIGN KEY(dni_2) REFERENCES director);
              
CREATE TABLE personaje(
              id_J          VARCHAR(7),
              nombre        VARCHAR(20),
              rol           VARCHAR(20)         CONSTRAINT        nn_rol_personaje            NOT NULL,
              descripcion   VARCHAR(200)        CONSTRAINT        nn_descripcion_personaje    NOT NULL,
              CONSTRAINT pk_personaje PRIMARY KEY(id_J, nombre),
              CONSTRAINT fk_id_J_personaje FOREIGN KEY(id_J) REFERENCES juego);
            
CREATE TABLE bug(
              id_B          VARCHAR(7),
              descrip   VARCHAR(200)         CONSTRAINT        nn_desc_bug         NOT NULL,
              version       VARCHAR(9),
              id_J1         VARCHAR(7),
              id_J2         VARCHAR(7),
              CONSTRAINT pk_bug PRIMARY KEY(id_B),
              CONSTRAINT fk_version_id_J1_bug FOREIGN KEY(version, id_J1) REFERENCES actualizacion,
              CONSTRAINT fk_id_J2_bug FOREIGN KEY(id_J2) REFERENCES juego);
              
CREATE TABLE ju_prog(
              id_J          VARCHAR(7),
              dni           VARCHAR(9),
              CONSTRAINT  pk_ju_prog  PRIMARY KEY(id_J, dni),
              CONSTRAINT fk_id_J_ju_prog  FOREIGN KEY(id_J) REFERENCES juego,
              CONSTRAINT fk_dni_ju_prog  FOREIGN KEY(dni) REFERENCES programador);
              
CREATE TABLE ju_dis(
              id_J          VARCHAR(7),
              dni           VARCHAR(9),
              CONSTRAINT pk_ju_dis PRIMARY KEY(id_J, dni),
              CONSTRAINT fk_id_J_ju_dis FOREIGN KEY(id_J) REFERENCES juego,
              CONSTRAINT fk_dni_ju_dis FOREIGN KEY(dni) REFERENCES diseñador);

SELECT e.dni, e.nombre, e.apellidos, e.f_nacimiento "FECHA DE NACIMIENTO", e.domicilio, e.cp "CÓDIGO POSTAL", e.nif "NIF DE EMPRESA" 
FROM empleado e, programador p
WHERE p.dni_1 = e.dni AND (SELECT COUNT(*)
                           FROM ju_prog jp
                           WHERE e.dni = jp.dni) = 0;

/*Mostrar los datos de aquellas empresas que hayan contratado entre 2 y 4 anuncios, así como el número de anuncios realizados*/
SELECT e.nif, e.nombre, COUNT(*) "NÚMERO DE ANUNCIOS REALIZADOS"
FROM empresa e, anuncio a
WHERE a.nif = e.nif
GROUP BY e.nif, e.nombre
HAVING COUNT(*) BETWEEN 2 AND 4;

/*Mostrar la información de los productos que ha anunciado la empresa XBOX GAME STUDIOS, así como la duración en minutos del anuncio redondeada al alza, el coste del anuncio y el nombre de la agencia que lo realizó*/
SELECT p.cod_pro "CÓDIGO DE PRODUCTO", p.nombre, p.descripcion, p.precio, ag.nombre "ANUNCIANTE", TO_CHAR(a.duracion/60, '9') "DURACIÓN DEL ANUNCIO", a.precio "CONSTE DEL ANUNCIO"
FROM empresa e, anuncio a, ag_em ae, agencia ag, producto p
WHERE e.nif = ae.nif AND ag.id_agencia = ae.id_agencia AND a.id_agencia = ae.id_agencia AND e.nif = ae.nif AND e.nif = p.nif AND p.cod_pro = a.cod_pro AND e.nombre = 'XBOX GAME STUDIOS';

/*Mostrar las parejas de empresas que han colaborado en el desarrollo de algún juego*/
SELECT e1.nombre, e2.nombre
FROM empresa e1, empresa e2
WHERE EXISTS(SELECT *
             FROM emp_emp em
             WHERE em.nif_1 = e1.nif AND em.nif_2 = e2.nif);
             
/*Mostrar el nombre y apellidos de aquellos programadores que solo han desarrollado juegos para su empresa, es decir, que no hayan trabajado en ninguna colaboración*/
SELECT e.nombre, e.apellidos, em.nombre "EMPRESA"
FROM programador p, empleado e, empresa em
WHERE p.dni_1 = e.dni AND em.nif = e.nif AND NOT EXISTS((SELECT j.id_juego
                                                         FROM ju_prog jp, juego j
                                                         WHERE j.id_juego = jp.id_J AND jp.dni = p.dni_1) MINUS (SELECT j.id_juego
                                                                                                                 FROM juego j
                                                                                                                 WHERE j.nif_1 = e.nif));
                                                                                                                 
/*Mostrar los datos del director o directores que más empleados tienen a su cargo*/
SELECT e.nombre, e.apellidos, e.dni, e.f_nacimiento "FECHA DE NACIMIENTO", e.domicilio, e.cp "CÓDIGO POSTAL", e.nif "NIF DE LA EMPRESA"
FROM empleado e, director d
WHERE e.dni = d.dni AND (SELECT COUNT(*)
                         FROM programador p
                         WHERE p.dni_2 = d.dni) + (SELECT COUNT(*)
                                                   FROM diseñador dis
                                                   WHERE dis.dni_2 = d.dni) = (SELECT MAX(COUNT(*))
                                                                               FROM programador p, director dir
                                                                               WHERE p.dni_2 = dir.dni
                                                                               GROUP BY dir.dni) + (SELECT MAX(COUNT(*))
                                                                                                    FROM diseñador dis, director dir
                                                                                                    WHERE dis.dni_2 = dir.dni
                                                                                                    GROUP BY dir.dni);

/*Actualizar el precio de los anuncios a la media de los precios de los anuncios realizados por la agencia con el identificador 1MA93JEPX*/
UPDATE anuncio a SET a.precio = (SELECT AVG(a.precio)
                                 FROM agencia ag
                                 WHERE ag.id_agencia = '1MA93JEPX');

/*Eliminar los anuncios realizados por la agencia con el identificador 83MF4J30B y que tengan una duración mayor a 100 segundos*/
DELETE FROM anuncio a 
WHERE a.duracion > 100 
  AND a.id_agencia = '83MF4J30B';

/*Mostrar la media de la cantidad de productos cuyo precio sea menor que 50*/
SELECT AVG(COUNT(pro.cod_pro))
FROM producto pro
WHERE pro.precio < 50
GROUP BY pro.cod_pro;

/*Mostrar la suma del precio de cada producto y la media de estos */
SELECT cod_pro, SUM(precio) "SUMA PRECIOS",TO_CHAR(AVG(precio),'999.99') "PRECIO MEDIO"
FROM producto
GROUP BY cod_pro
ORDER BY cod_pro ASC;


/*Mostrar la suma de precios de los anuncios creados por la empresa cuyo NIF es 12345678H*/
SELECT SUM(anu.precio) "Suma precios"
FROM anuncio anu, empresa emp
WHERE emp.nif = '12345678H'
  AND anu.nif = emp.nif

/*Mostrar el nuemero de empleados cuyo dni acaba en A*/

FROM empleado
WHERE empleado.dni LIKE '%A';
